65 /* Period parameters */
66 const int N = 624;
67 const int M = 397;
68 const unsigned int MATRIX_A = 0x9908b0dfU; /* constant vector a */
69 const unsigned int UPPER_MASK = 0x80000000U; /* most significant w-r bits */
70 const unsigned int LOWER_MASK = 0x7fffffffU; /* least significant r bits */
71 
72 static unsigned int mt[N]; /* the array for the state vector */
73 static int mti = N + 1; /* mti==N+1 means mt[N] is not initialized */
74 
75 /* initializes mt[N] with a seed */
76 void init_genrand(unsigned int s) {
77  mt[0] = s & 0xffffffffU;
78  for (mti = 1; mti < N; mti++) {
79  mt[mti] =
80  (1812433253UL * (mt[mti - 1] ^ (mt[mti - 1] >> 30)) + mti);
81  /* See Knuth TAOCP Vol2. 3rd Ed. P.106 for multiplier. */
82  /* In the previous versions, MSBs of the seed affect */
83  /* only MSBs of the array mt[]. */
84  /* 2002/01/09 modified by Makoto Matsumoto */
85  mt[mti] &= 0xffffffffU;
86  /* for >32 bit machines */
87  }
88 }

90 /* initialize by an array with array-length */
91 /* init_key is the array for initializing keys */
92 /* key_length is its length */
93 /* slight change for C++, 2004/2/26 */
94 void init_by_array(unsigned int init_key[], int key_length) {
95  int i, j, k;
96  init_genrand(19650218U);
97  i = 1; j = 0;
98  k = (N > key_length ? N : key_length);
99  for (; k; k--) {
100  mt[i] = (mt[i] ^ ((mt[i - 1] ^ (mt[i - 1] >> 30)) * 1664525UL))
101  + init_key[j] + j; /* non linear */
102  mt[i] &= 0xffffffffULL; /* for WORDSIZE > 32 machines */
103  i++; j++;
104  if (i >= N) { mt[0] = mt[N - 1]; i = 1; }
105  if (j >= key_length) j = 0;
106  }
107  for (k = N - 1; k; k--) {
108  mt[i] = (mt[i] ^ ((mt[i - 1] ^ (mt[i - 1] >> 30)) * 1566083941UL))
109  - i; /* non linear */
110  mt[i] &= 0xffffffffUL; /* for WORDSIZE > 32 machines */
111  i++;
112  if (i >= N) { mt[0] = mt[N - 1]; i = 1; }
113  }
114 
115  mt[0] = 0x80000000ULL; /* MSB is 1; assuring non-zero initial array */
116 }

118 /* generates a random number on [0,0xffffffff]-interval */
119 unsigned int genrand_int32(void) {
120  unsigned int y;
121  static unsigned int mag01[2] = { 0x0U, MATRIX_A };
122  /* mag01[x] = x * MATRIX_A for x=0,1 */
123 
124  if (mti >= N) { /* generate N words at one time */
125  int kk;
126 
127  if (mti == N + 1) /* if init_genrand() has not been called, */
128  init_genrand(5489UL); /* a default initial seed is used */
129 
130  for (kk = 0; kk < N - M; kk++) {
131  y = (mt[kk] & UPPER_MASK) | (mt[kk + 1] & LOWER_MASK);
132  mt[kk] = mt[kk + M] ^ (y >> 1) ^ mag01[y & 0x1UL];
133  }
134  for (; kk < N - 1; kk++) {
135  y = (mt[kk] & UPPER_MASK) | (mt[kk + 1] & LOWER_MASK);
136  mt[kk] = mt[kk + (M - N)] ^ (y >> 1) ^ mag01[y & 0x1UL];
137  }
138  y = (mt[N - 1] & UPPER_MASK) | (mt[0] & LOWER_MASK);
139  mt[N - 1] = mt[M - 1] ^ (y >> 1) ^ mag01[y & 0x1UL];
140 
141  mti = 0;
142  }
143 
144  y = mt[mti++];
145 
146  /* Tempering */
147  y ^= (y >> 11);
148  y ^= (y << 7) & 0x9d2c5680ULL;
149  y ^= (y << 15) & 0xefc60000ULL;
150  y ^= (y >> 18);
151 
152  return y;
153 }

178 /* generates a random number on [0,1) with 53-bit resolution*/
179 double genrand_res53(void) {
180  unsigned int a = genrand_int32() >> 5, b = genrand_int32() >> 6;
181  return(a*67108864.0 + b)*(1.0 / 9007199254740992.0);
