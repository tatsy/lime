84  Tensor operator*(Tensor t) const {
85  Tensor s;
86  for (int i = 0; i < 2; i++) {
87  for (int j = 0; j < 2; j++) {
88  s.data[(i << 1) + j] = 0.0;
89  for (int k = 0; k < 2; k++) {
90  s.data[(i << 1) + j] += data[(i << 1) + k] * t.data[(k << 1) + k];
91  }
92  }
93  }

95  }

117  double& elem(int i, int j) {

130  double det() const {

134  double norm() const {
135  double ret = 0.0;

137  ret += data[i] * data[i];

142  std::pair<double, double> eigen() const {
143  double p = -(data[0] + data[3]);
144  double q = det();
145  double D = sqrt(p * p - 4.0 * q);
146  return std::make_pair((-p + D) / 2.0, (-p - D) / 2.0);
