47 cv::Point2f generateRandomPointAround(const cv::Point2f& v, double min_dist) {
48  double radius = (1.0 + genrand_real1()) * min_dist;
49  double angle = 2.0 * PI * genrand_real1();
50  cv::Point2f ret;
51  ret.x = v.x + static_cast<int>(radius * cos(angle));
52  ret.y = v.y + static_cast<int>(radius * sin(angle));
53  return ret;
54 }

56 bool inNeighborhoodForList(const Grid<cv::Point2f>& grid, cv::Point2f point, double min_dist, double cellSize) {
57  cv::Point2f gridCoord = coord2Grid(point, cellSize);
58  int gridW = grid.cols();
59  int gridH = grid.rows();
60  for (int dy = -2; dy <= 2; dy++) {
61  for (int dx = -2; dx <= 2; dx++) {
62  int xx = static_cast<int>(gridCoord.x + dx);
63  int yy = static_cast<int>(gridCoord.y + dy);
64  if (grid.hasCell(yy, xx)) {
65  const std::vector<cv::Point2f>& v = grid(yy, xx);
66  if (!v.empty()) {
67  for (int i = 0; i < v.size(); i++) {
68  double deltaX = point.x - v[i].x;
69  double deltaY = point.y - v[i].y;
70  double dist = sqrt(deltaX * deltaX + deltaY * deltaY);
71  if (dist < min_dist) {
72  return true;
73  }
74  }
75  }
76  }
77  }
78  }
79  return false;
80 }

93 void pdsRandomQueue(std::vector<cv::Point2f>* points, const cv::Mat& grayImage, double min_radius, double max_radius) {
94  checkInputMat(grayImage);
95 
96  const int width = grayImage.cols;
97  const int height = grayImage.rows;
98  const int dim = grayImage.channels();
99  double cellSize = std::max(width, height) / 15.0;
100 
101  Random rand = Random::getRNG();
102 
103  cv::Point2f gridCoord;
104  int new_point_count = 8;
105  int gridW = static_cast<int>(ceil(width / cellSize));
106  int gridH = static_cast<int>(ceil(height / cellSize));
107  Grid<cv::Point2f> grid(gridH, gridW);
108  lime::random_queue<cv::Point2f> process;
109 
110  if (points->empty()) {
111  cv::Point2f firstPoint = cv::Point2f(rand.randInt(width), rand.randInt(height));
112  process.push(firstPoint);
113  points->push_back(firstPoint);
114  gridCoord = coord2Grid(firstPoint, cellSize);
115 
116  const int gx = static_cast<int>(gridCoord.x);
117  const int gy = static_cast<int>(gridCoord.y);
118  grid.pushAt(gy, gx, firstPoint);
119  } else {
120  for (int i = 0; i < points->size(); i++) {
121  cv::Point2f p = points->at(i);
122  double min_dist = minDistFromIntensity(p, grayImage, min_radius, max_radius);
123  if (p.x >= 0 && p.y >= 0 && p.x < width && p.y < height) {
124  if (!inNeighborhoodForList(grid, p, min_dist, cellSize)) {
125  process.push(p);
126  gridCoord = coord2Grid(p, cellSize);
127 
128  const int gx = static_cast<int>(gridCoord.x);
129  const int gy = static_cast<int>(gridCoord.y);
130  grid.pushAt(gy, gx, p);
131  } else {
132  points->erase(points->begin() + i);
133  i--;
134  }
135  }
136  }
137  }
138 
139  while (!process.empty()) {
140  cv::Point2f p = process.pop();
141  for (int i = 0; i < new_point_count; i++) {
142  double min_dist = minDistFromIntensity(p, grayImage, min_radius, max_radius);
143  cv::Point2f q = generateRandomPointAround(p, min_dist);
144  if (q.x >= 0 && q.y >= 0 && q.x < width && q.y < height) {
145  if (!inNeighborhoodForList(grid, q, min_dist, cellSize)) {
146  process.push(q);
147  points->push_back(q);
148  gridCoord = coord2Grid(q, cellSize);
149 
150  const int gx = static_cast<int>(gridCoord.x);
151  const int gy = static_cast<int>(gridCoord.y);
152  grid.pushAt(gy, gx, q);
153  }
154  }
155  }
156  }
157 } // function pdsRandomQueue

165 bool isConflict(const cv::Mat& gray, const cv::Mat& noise, cv::Point2f p, double min_radius, double max_radius) {
166  const int ksize = static_cast<int>(max_radius * 2.0);
167  const int width = gray.cols;
168  const int height = gray.rows;
169 
170  double r1 = minDistByIntensity(p, gray, min_radius, max_radius);
171  for (int dy = -ksize; dy <= ksize; dy++) {
172  for (int dx = -ksize; dx <= ksize; dx++) {
173  int xx = static_cast<int>(p.x + dx);
174  int yy = static_cast<int>(p.y + dy);
175  if (xx >= 0 && yy >= 0 && xx < width && yy < height) {
176  if (noise.at<float>(yy, xx) == 1.0f) {
177  double r2 = minDistByIntensity(cv::Point2f(xx, yy), gray, min_radius, max_radius);
178  double rmax = std::max(r1, r2);
179  if (dx * dx + dy * dy < rmax * rmax) {
180  return true;
181  }
182  }
183  }
184  }
185  }
186  return false;
187 }

189 bool throwSample(const cv::Mat& gray, const cv::Mat& noise, cv::Point2f* newPoint,
190  int nTrial, const cv::Rect& region, double min_radius, double max_radius) {
191  const int width = gray.cols;
192  const int height = gray.rows;
193 
194  for (int t = 0; t < nTrial; t++) {
195  int rx = region.x + genrand_int31() % region.width;
196  int ry = region.y + genrand_int31() % region.height;
197  if (rx >= 0 && ry >= 0 && rx < width && ry < height) {
198  cv::Point2f p = cv::Point2f(rx, ry);
199  if (!isConflict(gray, noise, p, min_radius, max_radius)) {
200  *newPoint = p;
201  return true;
202  }
203  }
204  }
205  return false;

224 void pdsParallel(std::vector<cv::Point2f>* points, const cv::Mat& grayImage, double min_radius, double max_radius) {
225  checkInputMat(grayImage);
226 
227  const int width = grayImage.cols;
228  const int height = grayImage.rows;
229  const int dim = grayImage.channels();
230 
231  // initialze noises
232  cv::Mat noise = cv::Mat::zeros(height, width, CV_32FC1);
233  if (!points->empty()) {
234  lime::random_queue<cv::Point2f> que;
235  const int np = static_cast<int>(points->size());
236  for (int i = 0; i < np; i++) {
237  que.push(points->at(i));
238  }
239 
240  while (!que.empty()) {
241  cv::Point2f p = que.pop();
242  int px = static_cast<int>(p.x);
243  int py = static_cast<int>(p.y);
244 
245  int rx = static_cast<int>(px - min_radius);
246  int ry = static_cast<int>(py - min_radius);
247  int rw = static_cast<int>(min_radius * 2.0);
248  int rh = static_cast<int>(min_radius * 2.0);
249 
250  cv::Rect region(rx, ry, rw, rh);
251  if (!containPoint(noise, region)) {
252  noise.at<float>(py, px) = 1.0f;
253  }
254  }
255  }
256 
257  // precompute
258  int nPhaseGroup = static_cast<int>(2.0 * sqrt(2.0) + 1.0);
259  int nPhaseGroup2 = nPhaseGroup * nPhaseGroup;
260  std::vector<int> order(nPhaseGroup2);
261  for (int i = 0; i < nPhaseGroup2; i++) {
262  order[i] = i;
263  }
264 
265  // determine traverse order for phase groups
266  for (int i = 0; i < 100; i++) {
267  int p = genrand_int31() % nPhaseGroup2;
268  int q = genrand_int31() % nPhaseGroup2;
269  int t = order[p];
270  order[p] = order[q];
271  order[q] = t;
272  }
273 
274  double limit = 2.0 * sqrt(2.0) * min_radius;
275  int gridW = width;
276  int gridH = height;
277  while (std::min(gridW, gridH) > limit) {
278  int nGridX = width / gridW;
279  int nGridY = height / gridH;
280  int cellW = gridW / nPhaseGroup + 1;
281  int cellH = gridH / nPhaseGroup + 1;
282 
283  ompfor (int gy = 0; gy < nGridY; gy++) { // NOLINT
284  for (int gx = 0; gx < nGridX; gx++) {
285  for (int k = 0; k < nPhaseGroup2; k++) {
286  int i = order[k];
287  int cx = i / nPhaseGroup;
288  int cy = i % nPhaseGroup;
289  int sx = gx * gridW + cx * cellW;
290  int sy = gy * gridH + cy * cellH;
291  cv::Rect omega = cv::Rect(sx, sy, cellW, cellH);
292  if (!containPoint(noise, omega)) {
293  cv::Point2f p;
294  if (throwSample(grayImage, noise, &p, 10, omega, min_radius, max_radius)) {
295  int ipx = static_cast<int>(p.x);
296  int ipy = static_cast<int>(p.y);
297  noise.at<float>(ipy, ipx) = 1.0f;
298  }
299  }
300  }
301  }
302  }
303  gridW /= 2;
304  gridH /= 2;
305  }
306 
307  // count number of points
308  points->clear();
309  for (int y = 0; y < height; y++) {
310  for (int x = 0; x < width; x++) {
311  if (noise.at<float>(y, x) == 1.0f) {
312  points->push_back(cv::Point2f(x, y));
313  }
314  }
315  }
316 } // function pdsParallel

320 void poissonDisk(cv::InputArray grayImage, std::vector<cv::Point2f>* samplePoints,
321  PdsMethod pdsMethod, double minRadius, double maxRadius) {
322  switch (pdsMethod) {
323  case PDS_RAND_QUEUE:
324  pdsRandomQueue(samplePoints, grayImage.getMat(), minRadius, maxRadius);
325  break;
326 
327  case PDS_FAST_PARALLEL:
328  pdsParallel(samplePoints, grayImage.getMat(), minRadius, maxRadius);
329  break;
330 
331  default:
332  Assertion(false, "Unknown method specified to Poisson disk sampling.");
333  }
334 }
